# 批量删除指定大小的文件
```shell
$ for i in $(ls ./);do if [ $(ls -l $i|awk '{print $5}') -lt 5000 ]; then rm $i;fi;done

# ls -l 查看文件属性 第5列为大小，单位字节
# -lt 5000 小于5000B 小于5KB
```
